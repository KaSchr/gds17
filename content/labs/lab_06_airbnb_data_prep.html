geos = gpd.GeoSeries(db[['longitude', 'latitude']]\
            .apply(lambda x: geoms.Point((x.longitude, x.latitude)), axis=1), \
            crs={'init': 'epsg:4326'})

db['geometry'] = geos
db = gpd.GeoDataFrame.from_records(db)
db.crs = geos.crs
